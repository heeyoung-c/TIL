# html-webpack-plugin
```js
const HtmlPlugin = require('html-webpack-plugin')

module.exports = {
  entry: './main.js',
  output: {
    clean: true
  },
  plugins: [
    new HtmlPlugin({
      template: './index.html' // 주변에 index.html파일을 복사해서 넣어라!
    }) // html을 넣어줄 플러그인
  ]    // html 파일에 script를 직접 연결할 필요가 없다 webpack이 알아서 넣어준다!
}
```
